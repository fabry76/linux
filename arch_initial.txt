#Set KB language
loadkeys it.map.gz

##Enable SSH from remote if needed
#Set root password
passwd

#Enable SSH service
systemctl start sshd.service

#Update the system clock
timedatectl set-ntp true

#View disks and partitions
lsblk

#Create 3 partitions
cfdisk /dev/nvme0n1

Device             Start       End   Sectors  Size Type
/dev/nvme0n1p1      2048   1050623   1048576  512M EFI System
/dev/nvme0n1p2   1050624 500117503 499066880  238G Linux filesystem

cryptsetup -y -v luksFormat /dev/nvme0n1p2
cryptsetup open /dev/nvme0n1p2 luks
pvcreate /dev/mapper/luks  *create the physical volume
vgcreate main /dev/mapper/luks   *create the volume group
lvcreate -L 100G main -n home
lvcreate -l 100%FREE main -n root
mkfs.ext4 /dev/mapper/main-home
mkfs.ext4 /dev/mapper/main-root
mkfs.fat -F32 /dev/nvme0n1p1
mount /dev/mapper/main-root /mnt
mount --mkdir /dev/mapper/main-home /mnt/home
mount --mkdir /dev/nvme0n1p1 /mnt/boot

#install base system
pacstrap /mnt base base-devel linux-lts linux-lts-headers linux-firmware nano intel-ucode lvm2

#Generate fstab
genfstab -U /mnt >> /mnt/etc/fstab

#Change root into the new system:
arch-chroot /mnt /bin/bash

#Set the time zone
ln -sf /usr/share/zoneinfo/Europe/Rome /etc/localtime

#synch hw clock to system clock
hwclock --systohc

#Locale 
nano /etc/locale.gen *(uncomment US and IT)
locale-gen
echo "LANG=en_US.UTF-8" | tee -a /etc/locale.conf

#hostname
echo "arco" | tee -a /etc/hostname

tee -a /etc/hosts  << END
127.0.0.1   localhost
::1         localhost
127.0.1.1   arco.localdomain    arco
END

#packages
pacman -S grub sbctl efibootmgr networkmanager xdg-user-dirs git

#netowork manager
systemctl enable NetworkManager

#mkinitcpio
nano /etc/mkinitcpio.conf
*HOOKS= after "block" type "encrypt" "lvm2"
mkinitcpio -p linux-lts

#grub
grub-install --target=x86_64-efi --efi-directory=/boot --bootloader-id=GRUB --modules="tpm" --disable-shim-lock
sed -i 's/GRUB_TIMEOUT=5/GRUB_TIMEOUT=2/g' /etc/default/grub
*blkid copy the UUID of the root partition /dev/nvme0n1p2
nano /etc/default/grub
*GRUB_CMDLINE_LINUX="cryptdevice=UUID=xxxxxxxx:luks root=/dev/mapper/main-root"
grub-mkconfig -o /boot/grub/grub.cfg

#users
useradd -m -G wheel fabri
passwd fabri
EDITOR=nano visudo
*uncomment line %wheel ALL=(ALL:ALL) ALL

#secure boot keys (RESET SECURE BOOT KEYS AND TPM FROM BIOS FIRST)
sbctl create-keys
sbctl enroll-keys -m
sbctl sign -s /boot/vmlinuz-linux-lts
sbctl sign -s /boot/EFI/GRUB/grubx64.efi

#enrolling the TPM
systemd-cryptenroll /dev/nvme0n1p2 --recovery-key
systemd-cryptenroll /dev/nvme0n1p2 --wipe-slot=empty --tpm2-device=auto
systemd-cryptenroll /dev/nvme0n1p3 --recovery-key
systemd-cryptenroll /dev/nvme0n1p3 --wipe-slot=empty --tpm2-device=auto

#exit and unmount
exit
umount -a
reboot
