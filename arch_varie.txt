#snap
git clone https://aur.archlinux.org/snapd.git
cd snapd
makepkg -si
sudo systemctl enable snapd.socket
sudo systemctl enable snapd.apparmor.service
sudo ln -s /var/lib/snapd/snap /snap
#reboot
sudo systemctl enable --now snapd.seeded.service
sudo snap install code --classic
sudo snap install onlyoffice-desktopeditors

#starship
tee -a .bashrc  << END
if [ ! -e /run/.toolboxenv ]; then
     eval "$(starship init bash)"
else
:
fi
END

cp ~/Git/linux/etc/starship.toml ~/.config

#distrobox
distrobox create -n debian -i debian:latest --hostname "$(uname -n)" --additional-packages "build-essential nano git net-tools curl neofetch"
distrobox enter debian
##locale
sed -i 's/# it_IT.UTF-8 UTF-8/it_IT.UTF-8 UTF-8/g' /etc/locale.gen
locale-gen

#chrome
wget -q -O - https://dl-ssl.google.com/linux/linux_signing_key.pub > linux_signing_key.pub
install -D -o root -g root -m 644 linux_signing_key.pub /etc/apt/keyrings/linux_signing_key.pub
sh -c 'echo "deb [arch=amd64 signed-by=/etc/apt/keyrings/linux_signing_key.pub] http://dl.google.com/linux/chrome/deb/ stable main" > /etc/apt/sources.list.d/google-chrome.list'
apt update && apt install google-chrome-stable
